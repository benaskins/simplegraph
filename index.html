SimpleGraph

Zero-dependency SVG graphs. No jQuery, no Raphael, no build step.
GitHub

v2.0

Minimal

simplegraph('#el', [20, 23, 23, 28, 28, 22, 25], days);

Line + fill + points

simplegraph('#el', data, days, {
  penColor: '#DC2626',
  fillUnderLine: true,
  drawPoints: true,
  minYAxisValue: 30,
  units: 'ºC'
});

Y-axis caption

simplegraph('#el', rain, days, {
  penColor: '#2563EB',
  leftGutter: 60,
  minYAxisValue: 10,
  yAxisCaption: 'Max Rainfall',
  units: 'mm'
});

Multi-series

simplegraph('#el', temp, days, {
  penColor: '#DC2626', leftGutter: 90,
  minYAxisValue: 10, yAxisCaption: 'Max Temp',
  units: 'ºC', lowerBound: -10, drawPoints: true
}).more(rain, {
  penColor: '#2563EB',
  minYAxisValue: 10, yAxisCaption: 'Max Rainfall',
  units: 'mm', lowerBound: 0
});

Bar chart

simplegraph('#el', data, days, {
  penColor: '#DC2626',
  drawBars: true,
  drawLine: false,
  drawPoints: false,
  barWidth: 24,
  minYAxisValue: 30,
  units: 'ºC'
});
